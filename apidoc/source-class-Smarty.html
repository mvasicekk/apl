 614:         if (is_array($tpl_var)) {

 624:                         }
 625:                     } else {
 626:                         $this->_tpl_vars[$_key][] = $_val;
 627:                     }
 628:                 }
 629:             }
 630:         } else {
 631:             if ($tpl_var != '' && isset($value)) {
 632:                 if(!@is_array($this->_tpl_vars[$tpl_var])) {
 633:                     settype($this->_tpl_vars[$tpl_var],'array');

 635:                 if($merge && is_array($value)) {
