3569:         /**
3570:          * Converts UTF-8 strings to UTF16-BE.<br>
3571:          * Based on: http://www.faqs.org/rfcs/rfc2781.html
3572:          * <pre>
3573:          *   Encoding UTF-16:
3574:          * 
3575:          *   Encoding of a single character from an ISO 10646 character value to
3576:          *    UTF-16 proceeds as follows. Let U be the character number, no greater
3577:          *    than 0x10FFFF.
3578:          * 
3579:          *    1) If U < 0x10000, encode U as a 16-bit unsigned integer and
3580:          *       terminate.
3581:          * 
3582:          *    2) Let U' = U - 0x10000. Because U is less than or equal to 0x10FFFF,
3583:          *       U' must be less than or equal to 0xFFFFF. That is, U' can be
3584:          *       represented in 20 bits.
3585:          * 
3586:          *    3) Initialize two 16-bit unsigned integers, W1 and W2, to 0xD800 and
3587:          *       0xDC00, respectively. These integers each have 10 bits free to
3588:          *       encode the character value, for a total of 20 bits.
3589:          * 
3590:          *    4) Assign the 10 high-order bits of the 20-bit U' to the 10 low-order
3591:          *       bits of W1 and the 10 low-order bits of U' to the 10 low-order
3592:          *       bits of W2. Terminate.
3593:          * 
3594:          *    Graphically, steps 2 through 4 look like:
3595:          *    U' = yyyyyyyyyyxxxxxxxxxx
3596:          *    W1 = 110110yyyyyyyyyy
3597:          *    W2 = 110111xxxxxxxxxx
3598:          * </pre>
3599:          * @param string $str string to process.
3600:          * @param boolean $setbom if true set the Byte Order Mark (BOM = 0xFEFF)
3601:          * @return string
3602:          * @access protected
3603:          * @author Nicola Asuni
3604:          * @since 1.53.0.TC005 (2005-01-05)
3605:          * @uses UTF8StringToArray
3606:          */
3607:         protected function UTF8ToUTF16BE($str, $setbom=true) {
3608:             if(!$this->isunicode) {
3609:                 return $str; // string is not in unicode
3610:             }
3611:             $outstr = ""; // string to be returned
3612:             $unicode = $this->UTF8StringToArray($str); // array containing UTF-8 unicode values
3613:             $numitems = count($unicode);
3614:             
3615:             if ($setbom) {
3616:                 $outstr .= "\xFE\xFF"; // Byte Order Mark (BOM)
3617:             }
3618:             foreach($unicode as $char) {
3619:                 if($char == 0xFFFD) {
3620:                     $outstr .= "\xFF\xFD"; // replacement character
3621:                 } elseif ($char < 0x10000) {
3622:                     $outstr .= chr($char >> 0x08);
3623:                     $outstr .= chr($char & 0xFF);
3624:                 } else {
3625:                     $char -= 0x10000;
3626:                     $w1 = 0xD800 | ($char >> 0x10);
3627:                     $w2 = 0xDC00 | ($char & 0x3FF); 
3628:                     $outstr .= chr($w1 >> 0x08);
3629:                     $outstr .= chr($w1 & 0xFF);
3630:                     $outstr .= chr($w2 >> 0x08);
3631:                     $outstr .= chr($w2 & 0xFF);
3632:                 }
3633:             }
3634:             return $outstr;
3635:         }
