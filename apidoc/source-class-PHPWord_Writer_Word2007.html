 57:     public function save($pFilename = null) {
 58:         if(!is_null($this->_document)) {
 59:             
 60:             // If $pFilename is php://output or php://stdout, make it a temporary file...
 61:             $originalFilename = $pFilename;
 62:             if(strtolower($pFilename) == 'php://output' || strtolower($pFilename) == 'php://stdout') {
 63:                 $pFilename = @tempnam('./', 'phppttmp');
 64:                 if($pFilename == '') {
 65:                     $pFilename = $originalFilename;
 66:                 }
 67:             }
 68:             
 69:             // Create new ZIP file and open it for writing
 70:             $objZip = new ZipArchive();
 71:             
 72:             // Try opening the ZIP file
 73:             if($objZip->open($pFilename, ZIPARCHIVE::OVERWRITE) !== true) {
 74:                 if($objZip->open($pFilename, ZIPARCHIVE::CREATE) !== true) {
 75:                     throw new Exception("Could not open " . $pFilename . " for writing.");
 76:                 }
 77:             }
 78:             
 79:             
 80:             $sectionElements = array();
 81:             $_secElements = PHPWord_Media::getSectionMediaElements();
 82:             foreach($_secElements as $element) { // loop through section media elements
 83:                 if($element['type'] != 'hyperlink') {
 84:                     $this->_addFileToPackage($objZip, $element);
 85:                 }
 86:                 $sectionElements[] = $element;
 87:             }
 88:             
 89:             $_hdrElements = PHPWord_Media::getHeaderMediaElements();
 90:             foreach($_hdrElements as $_headerFile => $_hdrMedia) { // loop through headers
 91:                 if(count($_hdrMedia) > 0) {
 92:                     $objZip->addFromString('word/_rels/'.$_headerFile.'.xml.rels', $this->getWriterPart('documentrels')->writeHeaderFooterRels($_hdrMedia));
 93:                     foreach($_hdrMedia as $element) { // loop through header media elements
 94:                         $this->_addFileToPackage($objZip, $element);
 95:                     }
 96:                 }
 97:             }
 98:             
 99:             $_ftrElements = PHPWord_Media::getFooterMediaElements();
100:             foreach($_ftrElements as $_footerFile => $_ftrMedia) { // loop through footers
101:                 if(count($_ftrMedia) > 0) {
102:                     $objZip->addFromString('word/_rels/'.$_footerFile.'.xml.rels', $this->getWriterPart('documentrels')->writeHeaderFooterRels($_ftrMedia));
103:                     foreach($_ftrMedia as $element) { // loop through footers media elements
104:                         $this->_addFileToPackage($objZip, $element);
105:                     }
106:                 }
107:             }
108:             
109:             
110:             
111:             $_cHdrs    = 0;
112:             $_cFtrs    = 0;
113:             $rID       = PHPWord_Media::countSectionMediaElements() + 6;
114:             $_sections = $this->_document->getSections();
115:             
116:             foreach($_sections as $section) {
117:                 $_header = $section->getHeader();
118:                 if(!is_null($_header)) {
119:                     $_cHdrs++;
120:                     $_header->setRelationId(++$rID);
121:                     $_headerCount = $_header->getHeaderCount();
122:                     $_headerFile = 'header'.$_headerCount.'.xml';
123:                     $sectionElements[] = array('target'=>$_headerFile, 'type'=>'header', 'rID'=>$rID);
124:                     $objZip->addFromString('word/'.$_headerFile, $this->getWriterPart('header')->writeHeader($_header));
125:                 }
126:                 
127:                 $_footer = $section->getFooter();
128:                 if(!is_null($_footer)) {
129:                     $_cFtrs++;
130:                     $_footer->setRelationId(++$rID);
131:                     $_footerCount = $_footer->getFooterCount();
132:                     $_footerFile = 'footer'.$_footerCount.'.xml';
133:                     $sectionElements[] = array('target'=>$_footerFile, 'type'=>'footer', 'rID'=>$rID);
134:                     $objZip->addFromString('word/'.$_footerFile, $this->getWriterPart('footer')->writeFooter($_footer));
135:                 }
136:             }
137:         
138:             // build docx file
139:             // Write dynamic files
140:             $objZip->addFromString('[Content_Types].xml', $this->getWriterPart('contenttypes')->writeContentTypes($this->_imageTypes, $this->_objectTypes, $_cHdrs, $_cFtrs));
141:             $objZip->addFromString('_rels/.rels', $this->getWriterPart('rels')->writeRelationships($this->_document));
142:             $objZip->addFromString('docProps/app.xml', $this->getWriterPart('docprops')->writeDocPropsApp($this->_document));
143:             $objZip->addFromString('docProps/core.xml', $this->getWriterPart('docprops')->writeDocPropsCore($this->_document));
144:             $objZip->addFromString('word/document.xml', $this->getWriterPart('document')->writeDocument($this->_document));
145:             $objZip->addFromString('word/_rels/document.xml.rels', $this->getWriterPart('documentrels')->writeDocumentRels($sectionElements));
146:             $objZip->addFromString('word/styles.xml', $this->getWriterPart('styles')->writeStyles($this->_document));
147:             
148:             // Write static files
149:             $objZip->addFile(PHPWORD_BASE_PATH . 'PHPWord/_staticDocParts/numbering.xml', 'word/numbering.xml');
150:             $objZip->addFile(PHPWORD_BASE_PATH . 'PHPWord/_staticDocParts/settings.xml', 'word/settings.xml');
151:             $objZip->addFile(PHPWORD_BASE_PATH . 'PHPWord/_staticDocParts/theme1.xml', 'word/theme/theme1.xml');

153:             $objZip->addFile(PHPWORD_BASE_PATH . 'PHPWord/_staticDocParts/fontTable.xml', 'word/fontTable.xml');
154:             
155:             
156:             // Close file
157:             if($objZip->close() === false) {
158:                 throw new Exception("Could not close zip file $pFilename.");
159:             }
160:

162:             if($originalFilename != $pFilename) {

173:     private function _chkContentTypes($src) {
174:         $srcInfo   = pathinfo($src);
175:         $extension = strtolower($srcInfo['extension']);
176:         if(substr($extension, 0, 3) == 'php') {
177:             $extension = 'php';
178:         }
179:         $_supportedImageTypes = array('jpg', 'jpeg', 'gif', 'png', 'bmp', 'tif', 'tiff', 'php');
180:         
181:         if(in_array($extension, $_supportedImageTypes)) {
182:             $imagedata = getimagesize($src);
183:             $imagetype = image_type_to_mime_type($imagedata[2]);
184:             $imageext = image_type_to_extension($imagedata[2]);
185:             $imageext = str_replace('.', '', $imageext);
186:             if($imageext == 'jpeg') $imageext = 'jpg';
187:             
188:             if(!in_array($imagetype, $this->_imageTypes)) {
189:                 $this->_imageTypes[$imageext] = $imagetype;
190:             }
191:         } else {
192:             if(!in_array($extension, $this->_objectTypes)) {
193:                 $this->_objectTypes[] = $extension;
194:             }
195:         }

225:         if(isset($element['isMemImage']) && $element['isMemImage']) {
226:             $image = call_user_func($element['createfunction'], $element['source']);
227:             ob_start();
228:             call_user_func($element['imagefunction'], $image);
229:             $imageContents = ob_get_contents();
230:             ob_end_clean();
231:             $objZip->addFromString('word/'.$element['target'], $imageContents);
232:             imagedestroy($image);
233:             
234:             $this->_chkContentTypes($element['source']);
235:         } else {
236:             $objZip->addFile($element['source'], 'word/'.$element['target']);
237:             $this->_chkContentTypes($element['source']);
238:         }
